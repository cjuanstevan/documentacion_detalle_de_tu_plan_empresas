/**
 * Representa una clase DeviceSpecRepository.
 * @class
 * @module DeviceSpecRepository
 */
const DeviceSpecLocalDataSource = require('../dataSource/DeviceSpecLocalDataSource');
const deviceSpecLocalDataSource = new DeviceSpecLocalDataSource();

module.exports = class DeviceSpecRepository {
  static instance;
  /**
   * Crea una instancia de DeviceSpecRepository.
   * @constructor
   * @returns {DeviceSpecRepository} La instancia de DeviceSpecRepository.
   */
  constructor() {
    if (DeviceSpecRepository.instance) {
      return DeviceSpecRepository.instance;
    } else {
      DeviceSpecRepository.instance = this;
    }
  }

  /**
   * Crea datos del dispositivo en el almacenamiento local utilizando la especificación del dispositivo proporcionada.
   * @param {Object} deviceSpec - Los datos de especificación del dispositivo.
   * @returns {any} El resultado de llamar al método createDeviceDataInStorage de la instancia de deviceSpecLocalDataSource.
   */
  createDeviceDataInStorageLocal(deviceSpec) {
    return deviceSpecLocalDataSource.createDeviceDataInStorage(deviceSpec);
  }

  /**
   * Recupera datos del dispositivo del almacenamiento local.
   * @returns {any} El resultado de llamar al método getDeviceDataInStorage de la instancia de deviceSpecLocalDataSource.
   */
  getDeviceDataInStorageLocal() {
    return deviceSpecLocalDataSource.getDeviceDataInStorage();
  }
}
