/**
 * Almacena datos del dispositivo en el almacenamiento y recupera datos del dispositivo del almacenamiento.
 * @module deviceDataStorage
 */

/**
 * Almacena los datos del dispositivo en el almacenamiento.
 * @param {string} key - La clave utilizada para identificar los datos del dispositivo almacenados.
 * @param {object} deviceSpec - La especificación del dispositivo que se almacenará en el almacenamiento.
 * @returns {boolean} - Devuelve verdadero si los datos del dispositivo se almacenaron correctamente en el almacenamiento; de lo contrario, es falso.
 */
function createDeviceDataInStorage(key, deviceSpec) {
  try {
    my.setStorage({
      key: key,
      data: deviceSpec
    });
    return true;
  } catch (error) {
    return false;
  }
}

/**
 * Recupera los datos del dispositivo del almacenamiento.
 * @param {string} key - La clave utilizada para identificar los datos del dispositivo almacenados.
 * @returns {object|Error} - Devuelve los datos del dispositivo si se recuperan correctamente del almacenamiento; de lo contrario, devuelve el objeto de error.
 */
function getDeviceDataInStorage(key) {
  try {
    let deviceInfo = my.getStorageSync({
      key
    });
    return deviceInfo.data;
  } catch (error) {
    return error;
  }
}

module.exports.createDeviceDataInStorage = createDeviceDataInStorage;
module.exports.getDeviceDataInStorage = getDeviceDataInStorage;
