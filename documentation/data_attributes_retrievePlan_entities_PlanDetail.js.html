/**
 * Representa un detalle del plan.
 * @class PlanDetail
 */
module.exports = class PlanDetail {
  /**
   * Crea una nueva instancia de PlanDetail.
   * @constructor
   * @param {string} planId - El ID del plan.
   * @param {string} planName - El nombre del plan.
   * @param {string} planDescription - La descripción del plan.
   * @param {string} mobileData - La cantidad de datos móviles incluidos en el plan.
   * @param {number} socialNetworks - Indica si las redes sociales están incluidas en el plan.
   * @param {string[]} socialNetworksList - La lista de redes sociales incluidas en el plan.
   * @param {string} voice - Indica la cantidad de llamadas de voz que están incluidas en el plan.
   * @param {string} sms - Indica la cantidad de mensajes SMS que están incluidos en el plan.
   */
  constructor(planId, planName, planDescription, mobileData, socialNetworks, socialNetworksList, voice, sms) {
    this.planId = planId;
    this.planName = planName;
    this.planDescription = planDescription;
    this.mobileData = mobileData;
    this.socialNetworks = socialNetworks;
    this.socialNetworksList = socialNetworksList;
    this.voice = voice;
    this.sms = sms;
  }
}
