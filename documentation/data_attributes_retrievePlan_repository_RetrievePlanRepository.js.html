const RetrievePlanRemoteDataSource = require('../dataSource/RetrievePlanRemoteDataSource');
const retrievePlanRemoteDataSource = new RetrievePlanRemoteDataSource();

/**
 * Una clase singleton que representa un RetrievePlanRepository.
 * @class RetrievePlanRepository
 */
module.exports = class RetrievePlanRepository {
  static instance;
  /**
   * Crea una instancia de RetrievePlanRepository.
   *
   * @constructor
   */
  constructor() {
    if (RetrievePlanRepository.instance) {
      return RetrievePlanRepository.instance;
    } else {
      RetrievePlanRepository.instance = this;
    }
  }

  /**
   * Recupera un plan por ID de cuenta de una fuente de datos remota.
   * @function
   * @param {object} userData - Los datos del usuario que contienen el ID de la cuenta.
   * @param {object} deviceData - Los datos del dispositivo que contienen el ID del dispositivo.
   * @param {object} request - Los datos de la solicitud que contienen el ID del plan.
   * @returns {Promise} - El plan recuperado de la fuente de datos remota.
   */
  async retrievePlanByAccountIdRemote(userData, deviceData, request) {
    return await retrievePlanRemoteDataSource.retrievePlanByAccountId(userData, deviceData, request);
  }
}
