/**
   * Envía una solicitud POST a una URL especificada con los datos de usuario y dispositivo proporcionados.
   * @param {string} url - La URL a la que enviar la solicitud POST.
   * @param {object} userData - Datos del usuario que se incluirán en la carga útil de la solicitud.
   * @param {object} deviceData - Datos del dispositivo que se incluirán en los encabezados de la solicitud.
   * @param {object} request - Datos de solicitud adicionales de tipo RetrievePlanRequest que se incluirán en la carga útil de la solicitud.
   * @returns {Promise} - Se resuelve con la respuesta si la solicitud tiene éxito, se rechaza con la respuesta de error si la solicitud falla.
   */
function postRetrievePlanRemote(url, userData, deviceData, request) {
  return new Promise((resolve, reject) => {
    my.request({
      url: url,
      method: "POST",
      dataType: "json",
      data: {
        data: {
          AccountId: request.accountId,
          accountIdHEader: "",
          alias: "",
          custcode: "",
          esEmpresas: request.isEnterprise,
          isZonaPublica: request.isPublicZone,
          LineOfBusiness: request.lineOfBusiness,
          selected: request.selected,
          token: deviceData['X-SESSION-ID'],
          tokenLine: "",
          valida: 0
        }
      },
      headers: deviceData,
      success: res => {
        resolve(res);
      },
      fail: res => {
        reject(res.data.response);
      }
    });
  });
}
