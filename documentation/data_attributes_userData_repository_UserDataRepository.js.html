/**
 * Representa una clase DeviceSpecRepository.
 * @class
 * @module UserDataRepository
 */
const UserDataSource = require("../dataSource/UserDataLocalDataSource");
const userDataSource = new UserDataSource();

module.exports = class UserDataRepository {
  static instance;
  /**
   * Crea una instancia de UserDataRepository.
   * @constructor
   * @returns {UserDataRepository} La instancia de UserDataRepository.
   */
  constructor() {
    if (UserDataRepository.instance) {
      return UserDataRepository.instance;
    } else {
      UserDataRepository.instance = this;
    }
  }

  /**
   * Crea datos de usuario en el almacenamiento local.
   * @param {Object} userDataLogin - Los datos del usuario se almacenarán en el almacenamiento local.
   * @returns {any} - El resultado del método createUserDataInStorage de la instancia userDataSource.
   */
  createUserDataInStorageLocal(userDataLogin) {
    return userDataSource.createUserDataInStorage(userDataLogin);
  }

  /**
   * Recupera datos del usuario del almacenamiento local.
   * @returns {any} - El resultado del método getUserDataInStorage de la instancia userDataSource.
   */
  getUserDataInStorageLocal() {
    return userDataSource.getUserDataInStorage();
  }
};
