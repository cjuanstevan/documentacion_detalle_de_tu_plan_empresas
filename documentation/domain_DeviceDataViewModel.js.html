const DeviceSpecRepository = require("../data/attributes/deviceData/repository/DeviceSpecRepository");
const DeviceSpec = require("../data/attributes/deviceData/entities/DeviceSpec");

/**
 * Crea datos del dispositivo en el almacenamiento.
 *
 * @param {string} deviceDataString - La cadena de datos del dispositivo.
 * @returns {boolean} Verdadero si los datos del dispositivo se almacenan correctamente en el almacenamiento, falso si se produce un error durante la operaci√≥n de almacenamiento.
 */
function createDeviceDataInStorage(deviceDataString) {
  const deviceSpec = new DeviceSpec(JSON.parse(deviceDataString));
  const deviceSpecRepository = new DeviceSpecRepository();
  const response = deviceSpecRepository.createDeviceDataInStorageLocal(deviceSpec)
  return response;
}

/**
 * Recupera datos del dispositivo del almacenamiento local usando la clase DeviceSpecRepository.
 * @returns {any} Los datos del dispositivo recuperados del almacenamiento local.
 */
function getDeviceDataInStorage() {
  const deviceSpecRepository = new DeviceSpecRepository();
  let response = deviceSpecRepository.getDeviceDataInStorageLocal();
  return response;
}

module.exports.createDeviceDataInStorage = createDeviceDataInStorage;
module.exports.getDeviceDataInStorage = getDeviceDataInStorage;
