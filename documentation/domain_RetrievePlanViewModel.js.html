const RetrievePlanRepository = require("../data/attributes/retrievePlan/repository/RetrievePlanRepository");
const RetrievePlanRequest = require("../data/attributes/retrievePlan/entities/RetrievePlanRequest");
const PlanDetail = require("../data/attributes/retrievePlan/entities/PlanDetail");

/**
 * Recupera los detalles del plan según el ID de la cuenta del usuario.
 * @async
 * @param {object} userData - Un objeto que contiene datos del usuario, incluido el ID de la cuenta, el estado de la empresa, el estado de la zona pública, la línea de negocio y el estado seleccionado.
 * @param {object} deviceData - Un objeto que contiene datos del dispositivo.
 * @returns {object} - Un objeto que contiene el estado de éxito y los datos detallados del plan. Si la solicitud tiene éxito, "result.success" será "true" y "result.data" será una instancia de "PlanDetail". Si hay un error, `result.data` contendrá el mensaje de error o los datos de respuesta.
 */
async function getPlanDetailByAccountId(userData, deviceData) {
  const retrievePlanRepository = new RetrievePlanRepository();
  let result = {
    success: false,
    data: "Ha ocurrido un error",
  };

  try {
    let request = new RetrievePlanRequest(
      userData.accountId,
      userData.isEnterprise,
      userData.isPublicZone,
      userData.lineOfBusiness,
      userData.selected
    );

    const response = await retrievePlanRepository.retrievePlanByAccountIdRemote(
      userData,
      deviceData,
      request
    );
    if (response.data.error === 0) {
      result.success = true;
      result.data = new PlanDetail(
        response.data.response.PlanID,
        response.data.response.PlanName,
        response.data.response.PlanDescription,
        response.data.response.textTotal,
        response.data.response.SocialNetworks,
        response.data.response.SocialNetworksList,
        response.data.response.voz,
        response.data.response.sms
      );
    } else {
      result.data = response.data.response;
    }
  } catch (error) {
    result.data = "Some Error"
  }
  return result;
}

module.exports.getPlanDetailByAccountId = getPlanDetailByAccountId;
